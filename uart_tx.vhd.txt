library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.NUMERIC_STD.ALL;

entity uart_tx is
    Port (
        clk         : in  STD_LOGIC; 
        reset       : in  STD_LOGIC;
        tx_start    : in  STD_LOGIC;
        tx_data     : in  STD_LOGIC_VECTOR (7 downto 0);
        tx_serial   : out STD_LOGIC;
        tx_busy     : out STD_LOGIC
    );
end uart_tx;

architecture Behavioral of uart_tx is
    type state_type is (ST_IDLE, ST_START, ST_DATA, ST_STOP);
    signal state : state_type := ST_IDLE;
    
    signal data_buffer : STD_LOGIC_VECTOR(7 downto 0) := (others => '0');
    signal bit_index   : integer range 0 to 7 := 0;
begin
    process(clk, reset)
    begin
        if reset = '1' then
            state <= ST_IDLE;
            tx_serial <= '1';
            tx_busy <= '0';
            bit_index <= 0;
        elsif rising_edge(clk) then
            case state is
                when ST_IDLE =>
                    tx_serial <= '1';
                    tx_busy <= '0';
                    if tx_start = '1' then
                        data_buffer <= tx_data;
                        tx_busy <= '1';
                        state <= ST_START;
                    end if;
                when ST_START =>
                    tx_serial <= '0';
                    state <= ST_DATA;
                    bit_index <= 0;
                when ST_DATA =>
                    tx_serial <= data_buffer(bit_index);
                    if bit_index = 7 then
                        state <= ST_STOP;
                    else
                        bit_index <= bit_index + 1;
                    end if;
                when ST_STOP =>
                    tx_serial <= '1';
                    state <= ST_IDLE;
            end case;
        end if;
    end process;
end Behavioral;
